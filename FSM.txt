enum RoverState 
{
    INIT,
    SENSOR_CALIB,
    NAVIGATION,
    SAMPLE_PICK,
    DATA_UPLOAD,
    IDLE
};

class RoverFSM 
{
private:
    RoverState state;
    Sensors sensors;

public:
    RoverFSM() : state(INIT), sensors(10) {}

    void roverTask() 
    {
       
        sensors.sensorTask();

        switch(state) 
    {
            case INIT:
                break;

            case NAVIGATION:
                // move, detect sample/checkpoint
                break;

            case SAMPLE_PICK:
                // pick sample
                state = NAVIGATION;
                break;

            case DATA_UPLOAD:
                // send telemetry
                state = IDLE;
                break;

            case IDLE:
                // mission complete
                break;
        }
    }
};
